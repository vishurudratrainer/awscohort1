#!/bin/bash

# Update packages
yum update -y

# Install Python 3 and Pip
yum install -y python3 python3-pip

# Ensure we are using the latest pip
python3 -m pip install --upgrade pip

# Use 'python3 -m pip' to ensure libraries go to the correct site-packages
python3 -m pip install pandas numpy requests boto3

# Log the result
echo "Setup complete" > /home/ec2-user/setup_status.txt
